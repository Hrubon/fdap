\def\@makechapterhead#1{
  {\parindent \z@ \raggedright \normalfont
   \Huge\bfseries \@chapapp\space \thechapter \\ \medskip #1
   \par\nobreak
   \vskip 20\p@
}}

  {\parindent \z@ \raggedright \normalfont
   \Huge\bfseries #1

}}
